const { app } = require("@azure/functions"); const { TableClient } = require("@azure/data-tables"); function getClient() { const connectionString = process.env.AzureWebJobsStorage; const tableName = process.env.TABLE_NAME || "chatlogs"; return TableClient.fromConnectionString(connectionString, tableName); } app.http("getTopQuestions", { methods: ["GET", "OPTIONS"], authLevel: "anonymous", handler: async (request, context) => { const allowOrigin = "https://wilokorea.github.io"; if (request.method === "OPTIONS") { return { status: 204, headers: { "Access-Control-Allow-Origin": allowOrigin, "Access-Control-Allow-Methods": "GET, OPTIONS", "Access-Control-Allow-Headers": "Content-Type" } }; } try { const client = getClient(); const url = new URL(request.url); const category = url.searchParams.get("category") || "all"; const counter = new Map(); for await (const entity of client.listEntities()) { const entityCategory = entity.category || "all"; if (category !== "all" && entityCategory !== category) { continue; } const question = entity.question || ""; if (!question) continue; const matchedWeight = entity.matched === false ? 1 : 2; const sourceWeight = entity.source === "faq-click" ? 3 : 1; const score = matchedWeight * sourceWeight; if (!counter.has(question)) { counter.set(question, { question, category: entityCategory, score: 0 }); } counter.get(question).score += score; } const items = Array.from(counter.values()) .sort((a, b) => b.score - a.score) .slice(0, 4) .map(item => ({ question: item.question, category: item.category })); return { status: 200, jsonBody: { items }, headers: { "Access-Control-Allow-Origin": allowOrigin } }; } catch (err) { context.error(err); return { status: 500, jsonBody: { ok: false, message: "top question read failed" }, headers: { "Access-Control-Allow-Origin": allowOrigin } }; } } });
